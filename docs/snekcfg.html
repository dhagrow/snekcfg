snekcfg

snekcfg is a minimalist wrapper over configparser that offers a more streamlined user experience.

Expected options and their types are predefined to prevent errors and minimize repetitive logic within your codebase. This allows for simple, direct access to your configuration within your program without worrying about typos or type conversion.

example.py

import snekcfg

cfg = snekcfg.Config('example.cfg')

sct = cfg.section('server')
# the default value is used to automatically type as an int
sct.define('port', 8080)
# or, you can be explicit
sct.define('host', default='127.0.0.1', type=str)

# some common types like set[str] are already built-in,
# but here is an example of adding a codec for a new type.
# *type* can be a type object, or just a string like 'str_set'
cfg.register_type(
    type=set[str],
    encode=lambda v: ','.join(v),
    decode=lambda v: set(v.split(',')))

# sections can be accessed using dot notation (one level deep)
cfg.define('users.whitelist', default=set(), type=set[str])

# update values with dot notation
users = {'graham', 'john', 'terryg', 'eric', 'terryj', 'michael'}
cfg['users.whitelist'].update(users)
# or through the section
cfg.section('server')['port'] = 1337

# write to 'example.cfg'
cfg.write()

# let's tweak the config file externally
with open('example.cfg', 'r+') as f:
    s = f.read()
    f.seek(0)
    f.write(s.replace('1337', '1234'))

# read from 'example.cfg'
cfg.read()

# types are preserved
assert cfg['server.port'] == 1234
assert cfg['users.whitelist'] == users, users

229class INIFormat(Format):
230    """Support for reading/writing INI files using `configparser`.
231
232    *parser* can be used to pass in a custom `configparser.ConfigParser`
233        instance. The default instance disables interpolation.
234    """
235    def __init__(self, parser=None):
236        self._parser = parser or configparser.ConfigParser(interpolation=None)
237
238    def read(self, file, config):
239        parser = self._parser
240
241        parser.read_file(file)
242
243        for section, options in parser.items():
244            sct = config.section(section)
245            for name, value in options.items():
246                try:
247                    sct.set(name, value, decode=True)
248                except UnknownOption:
249                    log.warning('unknown option: %s', name)
250                    continue
251
252        self._clear()
253
254    def write(self, file, config):
255        parser = self._parser
256
257        for sct in config:
258            parser.add_section(sct.name)
259            for name, value in sct.items(encode=True):
260                parser[sct.name][name] = value
261
262        parser.write(file)
263
264        self._clear()
265
266    def _clear(self):
267        parser = self._parser
268        for sct in parser.sections():
269            parser.remove_section(sct)

310class StringCodec(Codec):
311    def register_default_types(self):
312        self.register_type(str, None, None)
313        self.register_type(int, str, int)
314        self.register_type(float, str, float)
315
316        _boolean_states = {'1': True, 'yes': True, 'true': True, 'on': True,
317            '0': False, 'no': False, 'false': False, 'off': False}
318        self.register_type(bool,
319            lambda x: 'true' if x else 'false',
320            lambda x: _boolean_states[x.lower()])
321
322        def clean_split(v):
323            yield from filter(None, (x.strip() for x in v.split(',')))
324
325        self.register_type('set[str]',
326            lambda v: ','.join(v),
327            lambda v: set(clean_split(v)))
328        self.register_type('list[str]',
329            lambda v: ','.join(v),
330            lambda v: list(clean_split(v)))
331        self.register_type('tuple[str, ...]',
332            lambda v: ','.join(v),
333            lambda v: tuple(clean_split(v)))
334
335        self.register_type('tuple[int, ...]',
336            lambda v: ','.join(str(x) for x in v),
337            lambda v: tuple(int(x) for x in clean_split(v)))

231    """Support for reading/writing INI files using `configparser`.
232
233    *parser* can be used to pass in a custom `configparser.ConfigParser`
234        instance. The default instance disables interpolation.
235    """
236    def __init__(self, parser=None):
237        self._parser = parser or configparser.ConfigParser(interpolation=None)
238
239    def read(self, file, config):
240        parser = self._parser
241
242        parser.read_file(file)
243
244        for section, options in parser.items():
245            sct = config.section(section)
246            for name, value in options.items():
247                try:
248                    sct.set(name, value, decode=True)
249                except UnknownOption:
250                    log.warning('unknown option: %s', name)
251                    continue
252
253        self._clear()
254
255    def write(self, file, config):
256        parser = self._parser
257
258        for sct in config:
259            parser.add_section(sct.name)
260            for name, value in sct.items(encode=True):
261                parser[sct.name][name] = value
262
263        parser.write(file)
264
265        self._clear()
266
267    def _clear(self):
268        parser = self._parser
269        for sct in parser.sections():
270            parser.remove_section(sct)
